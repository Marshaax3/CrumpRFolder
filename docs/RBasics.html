    if(n==2){
      TRUE
    } else if (any(n%%2:(n-1)==0)){
      FALSE
    } else {
      TRUE
    }
  }
PrimeNumbers <- Vectorize(PrimeNumbers)

  AllSum<-0
  for(i in m){
    AllSum <- AllSum+i
  }
return<-c(AllSum/length(m))
